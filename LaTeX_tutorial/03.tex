\documentclass{article}

% Using Packages in LaTeX
% Packages must be included in the preamble

\usepackage{amsmath} % Required for equation*

\begin{document}
    % equation without * puts numbering for the equation
    \begin{equation}
        f(x) = x^2
    \end{equation}

    % equation with * removes numbering for the equation
    \begin{equation*}
        f(x) = x^2
    \end{equation*}
\end{document}
